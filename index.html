    <script>

        function calculateConversion() {

            const dose = parseFloat(document.getElementById('dose').value);

            const currentOpioid = document.getElementById('current').value;

            const targetOpioid = document.getElementById('target').value;


            if (isNaN(dose) || dose <= 0) {

                document.getElementById('result').textContent = "Please enter a valid dose.";

                return;

            }


            // Conversion factors (simplified example)

            const conversionFactors = {

                morphine: 1,

                oxycodone: 1.5,

                hydromorphone: 4

            };


            const morphineEquivalent = dose / conversionFactors[currentOpioid];

            const targetDose = morphineEquivalent * conversionFactors[targetOpioid];


            document.getElementById('result').textContent = `The equivalent dose of ${targetOpioid} is ${targetDose.toFixed(2)} mg.`;

        }

    </script>
